\section{$B_c$ physics @ LHCb}

\frame{
   \frametitle{$B_c^+$ physics @ LHCb}
   \vspace{-0.6cm}
   \begin{itemize}
     \item Lowest bound state of $\bar{b}$ and $c$ quarks. Unlike $\bar{b}b$, $\bar{c}c$; decays {\em only} weakly. Unique, flavor asymmetry: $t$ is too heavy.
     %\item Produced mostly via $gg \to B_c^+ b \bar{c}$ at hadron colliders. Difficult at $e^+e^-$ at current lumis (super $Z$-factory?).
     %\item Discovered at \href{http://arxiv.org/abs/hep-ex/9805034}{CDF (1998)} via $B_c^+\to \jpsi \ell^+\nu$, \href{https://arxiv.org/abs/0712.1506}{CDF~II}/\href{https://arxiv.org/abs/0802.4258}{D0}/\href{http://arxiv.org/abs/1209.5634}{LHCb} via $B_c^+\to \jpsi \pip$.
   \end{itemize}
   \begin{columns}
     \column{0.45\textwidth}
     \begin{itemize}
     \item Either the $\bar{b}$ or $c$ can decay:\\
         \vspace{0.1cm}
         \small
         (a) \textcolor{blue}{$\bar{b}\to c W^+$}: $\jpsi \ellp \nu$, $\jpsi\pip$\\
         \vspace{0.1cm}
         (b) \textcolor{blue}{$c\to s W^+$}: $B_s\ellp \nu$, $B_s\pip$\\
         \vspace{0.1cm}
         (c) Weak \textcolor{red}{annihilation} (\textcolor{red}{WA}), \phantom{blab}\textcolor{red}{$\bar{b}c\to W^+$}: $\tau^+ \nu_\tau$, $\bar{K}^{\ast 0}\Kp$
     \end{itemize}
     \column{0.57\textwidth}
     \centering
     \includegraphics[width=2.65in]{figs/bc/feynman_diag.png}
   \end{columns}
   \vspace{0.2cm}
   \begin{itemize}
     \item Experimentally, $b\to c$ is clearer (due to the \jpsi) but Cabibbo suppressed.
     \vspace{0.2cm}
     \item $c\to s$ is challenging but Cabibbo favored ($\mathcal{B}\sim 70\%$). First seen in \textcolor{blue}{$B_c^+\to B_s^0\pip$} by \href{http://arxiv.org/abs/1308.4544}{LHCb} (\textcolor{blue}{1308.4544})     
     \vspace{0.2cm}
     \item \textcolor{red}{WA} is sensitive to \textcolor{red}{$H^+$}. CF over $B^+$ by $|\frac{V_{cb}}{V_{ub}}|^2\sim 100$
   \end{itemize}
}

\frame{
   \frametitle{WA studies via charmless $B_c\to h^+ h^- h'$}
     \vspace{-0.6cm}
   \begin{itemize}
     \item $B_c$ decays to charmless states proceeds only via \textcolor{red}{$\bar{b}c \to W^+ \to u \{d,s\}$}. $|V_{ud}|>|V_{us}|$ implies $\Delta S=0$ dominates over $\Delta S=1$. 
   \end{itemize}
   \begin{columns}
     \column{0.47\textwidth}
   \begin{itemize}
     \vspace{-0.85cm}
     \item LHCb is studying $ B_c \to \{KKK,\pi\pi\pi,KK\pi, p\bar{p}K,p \bar{p}\pi\}$.
     \vspace{0.01cm}
     \item \textcolor{red}{Large} available \textcolor{red}{phase-space}!
     \vspace{0.3cm}
     \item Intermediate $\Dz$, $B_{s,d}$ and $c\bar{c}$ also contribute.
   \end{itemize}
     \column{0.53\textwidth}
     \centering
     \includegraphics[width=2.5in]{figs/bc/bc_1.pdf}
   \end{columns}
   \begin{itemize}
     \item Some theory \href{http://arxiv.org/abs/0907.2256}{\textcolor{blue}{(0907.2256)}} estimates for (quasi) 2-body decays. \textcolor{red}{$\mathcal{B}<10^{-6}$}. Expect few events in Run~I. 
     \vspace{0.2cm}
     \item Utlimately, Dalitz analysis, $\mathcal{A}_{CP}$, ...
   \end{itemize}
}

\frame{
   \frametitle{Search for $B_c^+\to p\bar{p}\pip$ (NEW!) \href{http://arxiv.org/abs/1603.07037}{\scriptsize LHCb-PAPER-2016-001}}
     \vspace{-0.2cm}
    \begin{itemize}
     \item $m(p \bar{p})<2.85$~GeV: near-threshold $p \bar{p}$ production expected. Also $[2.85,3.15]$~GeV for the $\jpsi$ region.
    \end{itemize}

   \begin{columns}
     \column{0.68\textwidth}
     \vspace{-0.8cm}
    \begin{itemize}
     \item Measured $R_p\equiv \frac{f_c}{f_u}\times \mathcal{B}(B_c^+ \to p\bar{p}\pip)$, with \href{http://arxiv.org/pdf/1407.5907v1.pdf}{$B^+ \to p\bar{p}\pip$} as norm. mode
     \vspace{0.2cm}
     \item \textcolor{red}{No signal} detected. UL's for $95\%$ CL set as: $\textcolor{red}{R_p<3.6\times 10^{-8}}$ and $\textcolor{red}{R_p^{\jpsi} <8.4\times 10^{-6}}$. 
     \vspace{0.2cm}
     \item Compatible with \href{http://arxiv.org/abs/1411.2943}{LHCb $\mathcal{B}(B_c^+\to \jpsi \pip)$}. Hoping for some signal with Run~II data.
     \vspace{0.3cm}
     \item \textcolor{red}{$B_c^+ \to \Kp \Km \pip$} in the immediate pipeline.
    \end{itemize}
