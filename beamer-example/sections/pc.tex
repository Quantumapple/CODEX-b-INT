     \begin{itemize}
       \item Test \textcolor{red}{null hypothesis $H_0$}: no $P_c$'s are needed and $\Lambda^\ast$ resonances can alone explain data.

       \vspace{0.2cm}
       \item Signature of each spin-$J$ $\Lambda^{\ast J} \to pK$: \textcolor{blue}{moments} expansion in \textcolor{red}{$\cos \theta_{\Lambda^\ast}$}. 
     \end{itemize}

       \textcolor{blue}{\langle P^U_l \rangle}^k &= \displaystyle \sum_{i=0}^{n_{\rm cand}^k} (w_i/\epsilon_i) P_l(\textcolor{red}{\cos \theta_{\Lambda^\ast}^i}) \nonumber

       \item For a given $m_{pK}$ bin, limit $J_{\rm max}$.

       \vspace{0.15cm}
       \item $\Lambda^\ast$-only model \textcolor{red}{$H_0$}: \textcolor{red}{$l_{\rm max}(m_{pK})$}.
